    /**
 * Finds both minimum and maximum in an array.
 * @module order-statistics/minmax
 * @type {Function}
 * @param {Array} list
 * @param {compare} compare compares non-numbers
 * @returns {Array} [min, max]
 */
module.exports = (list, compare = (a, b) => a < b ) => {
  const l = list.length

  if (!l) return {min: null, max: null}

  if (l === 1) return {min: list[0], max: list[0]}

  return list.reduce((acc, curr, idx, arr) => {
    if (idx & 1) {
      let prev = arr[idx - 1]

      console.log(prev, curr)

      if (compare(prev, curr)) {
        if (compare(prev, acc.min)) acc.min = prev
        if (compare(acc.max, curr)) acc.max = curr
      } else {
        if (compare(acc.max, prev)) acc.max = prev
        if (compare(curr, acc.min)) acc.min = curr
      }
    }

    return acc
  }, { // accumulator
    min: list[0],
    max: list[0]
  })
}
